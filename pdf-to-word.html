        function handleFiles(e) {

            const selectedFiles = e.target.files;

            handleFileList(selectedFiles);

        }


        function handleFileList(fileList) {

            for (let file of fileList) {

                if (file.type === 'application/pdf') {

                    addFile(file);

                } else {

                    alert(`文件 ${file.name} 不是PDF格式，已跳过`);

                }

            }

            fileInput.value = '';

        }


        function addFile(file) {

            const fileId = Date.now() + Math.random();

            const fileInfo = {

                id: fileId,

                file: file,

                name: file.name,

                size: formatFileSize(file.size),

                status: '等待转换',

                progress: 0

            };

            files.push(fileInfo);

            renderFileList();

        }


        function formatFileSize(bytes) {

            if (bytes === 0) return '0 Bytes';

            const k = 1024;

            const sizes = ['Bytes', 'KB', 'MB', 'GB'];

            const i = Math.floor(Math.log(bytes) / Math.log(k));

            return parseFloat((bytes / Math.pow(k, i)).toFixed(2)) + ' ' + sizes[i];

        }

                        </div>

                    </div>

                    <div class="file-status">

                        <div class="progress-bar">

                            ${file.status}

                        </span>

                        ${file.status === '转换完成' ? 

                            `<button class="btn" style="padding: 8px 15px;" onclick="downloadFile('${file.id}')">

                                <i class="fas fa-download"></i> 下载

                            </button>` : 

                            file.status === '等待转换' ?

                            `<button class="btn" style="padding: 8px 15px; background: #ef4444;" onclick="removeFile('${file.id}')">

                                <i class="fas fa-times"></i> 移除

                            </button>` : ''

                        }

                    </div>

                </div>

            `).join('');

        }


        function removeFile(fileId) {

            files = files.filter(f => f.id !== fileId);

            renderFileList();

        }


        function clearAll() {

            if (converting) {

                alert('请等待当前转换完成');

                return;

            }

            files = [];

            renderFileList();

        }


        // 转换功能

        async function convertAll() {

            if (converting) return;

            

            converting = true;

            convertBtn.innerHTML = '<i class="fas fa-spinner fa-spin"></i> 转换中...';

            convertBtn.disabled = true;


            for (let fileInfo of files) {

                if (fileInfo.status !== '等待转换') continue;

                

                try {

                    updateFileStatus(fileInfo.id, '转换中', 0);

                    

                    // 模拟转换进度

                    for (let i = 0; i <= 100; i += 10) {

                        await new Promise(resolve => setTimeout(resolve, 200));

                        updateFileStatus(fileInfo.id, '转换中', i);

                    }

                    

                    // 这里调用实际的转换函数

                    const docxData = await convertPDFToWord(fileInfo.file);

                    

                    // 保存转换后的文件

                    fileInfo.docxData = docxData;

                    updateFileStatus(fileInfo.id, '转换完成', 100);

                    

                } catch (error) {

                    console.error('转换失败:', error);

                    updateFileStatus(fileInfo.id, '转换失败', 0);

                }

            }

            

            converting = false;

            convertBtn.innerHTML = '<i class="fas fa-sync-alt"></i> 开始转换所有文件';

            convertBtn.disabled = false;

        }


        function updateFileStatus(fileId, status, progress) {

            const fileIndex = files.findIndex(f => f.id === fileId);

            if (fileIndex !== -1) {

                files[fileIndex].status = status;

                files[fileIndex].progress = progress;

                renderFileList();

            }

        }


        // 实际转换函数 - 这里使用pdf-lib和docx库

        async function convertPDFToWord(pdfFile) {

            return new Promise(async (resolve, reject) => {

                try {

                    // 读取PDF文件

                    const arrayBuffer = await pdfFile.arrayBuffer();

                    

                    // 使用pdf-lib提取文本（实际项目中可能需要更复杂的解析）

                    const pdfDoc = await PDFLib.PDFDocument.load(arrayBuffer);

                    const pages = pdfDoc.getPages();

                    

                    // 创建Word文档结构

                    const doc = {

                        sections: [{

                            properties: {},

                            children: []

                        }]

                    };

                    

                    // 提取文本内容

                    for (let i = 0; i < pages.length; i++) {

                        const page = pages[i];

                        // 这里简化处理，实际需要更复杂的文本提取

                        const textContent = await page.getTextContent();

                        

                        if (textContent && textContent.items) {

                            // 添加页面标题

                            doc.sections[0].children.push({

                                type: 'paragraph',

                                children: [{

                                    type: 'text',

                                    text: `第 ${i + 1} 页`,

                                    bold: true

                                }]

                            });

                            

                            // 添加文本内容

                            for (const item of textContent.items) {

                                doc.sections[0].children.push({

                                    type: 'paragraph',

                                    children: [{

                                        type: 'text',

                                        text: item.str || ''

                                    }]

                                });

                            }

                            

                            // 添加分页符（最后一页除外）

                            if (i < pages.length - 1) {

                                doc.sections[0].children.push({

                                    type: 'pageBreak'

                                });

                            }

                        }

                    }

                    

                    // 生成Word文件（这里简化，实际应该生成完整的docx）

                    const blob = new Blob(['模拟的Word文件内容 - 实际需要完整的DOCX生成'], 

                        { type: 'application/vnd.openxmlformats-officedocument.wordprocessingml.document' });

                    

                    resolve(blob);

                    

                } catch (error) {

                    console.error('转换错误:', error);

                    reject(error);

                }

            });

        }


        // 下载文件

        function downloadFile(fileId) {

            const fileInfo = files.find(f => f.id === fileId);

            if (!fileInfo || !fileInfo.docxData) return;

            

            const url = URL.createObjectURL(fileInfo.docxData);

            const a = document.createElement('a');

            a.href = url;

            a.download = fileInfo.name.replace('.pdf', '.docx');

            document.body.appendChild(a);

            a.click();

            document.body.removeChild(a);

            URL.revokeObjectURL(url);

        }
